Table Relationships

  • GameNight.GN_ID → Game.GN_ID One GameNight has exactly 5 Games
  • Game.Game_ID → Cards.Game_ID One Game has many Cards (one per player)
  • Songs.Song_ID → Cards.Sq_B1 … Sq_O5 Each of the 24 active squares holds a Song_ID
  • Game.Game_WinnerCard_ID → Cards.Card_ID Optional back-reference to the winning card

-- ============================================================
-- MUSIC BINGO — DDL
-- ============================================================

CREATE TABLE GameNight (
  GN_ID          INT           NOT NULL AUTO_INCREMENT PRIMARY KEY,
  GN_Name        VARCHAR(100)  NOT NULL,
  GN_Date        DATE          NOT NULL,
  GN_Venue       VARCHAR(150)  NULL,
  GN_HostName    VARCHAR(100)  NULL,
  GN_MaxPlayers  INT           NULL,
  GN_Notes       TEXT          NULL,
  GN_IsActive    TINYINT(1)    NOT NULL DEFAULT 1,
  GN_CreatedAt   DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Songs (
  Song_ID           INT           NOT NULL AUTO_INCREMENT PRIMARY KEY,
  Song_Title        VARCHAR(200)  NOT NULL,
  Song_Artist       VARCHAR(150)  NOT NULL,
  Song_Year         SMALLINT      NULL,
  Song_Genre        VARCHAR(80)   NULL,
  Song_Album        VARCHAR(200)  NULL,
  Song_Duration_Sec SMALLINT      NULL,
  Song_IsActive     TINYINT(1)    NOT NULL DEFAULT 1,
  Song_CreatedAt    DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Game (
  Game_ID           INT          NOT NULL AUTO_INCREMENT PRIMARY KEY,
  GN_ID             INT          NOT NULL,
  Game_Number       TINYINT      NOT NULL,  -- 1 through 5
  Game_Name         VARCHAR(100) NULL,
  Game_StartTime    DATETIME     NULL,
  Game_EndTime      DATETIME     NULL,
  Game_WinPattern   VARCHAR(50)  NULL,
  Game_WinnerCard_ID INT         NULL,
  Game_Status       VARCHAR(20)  NOT NULL DEFAULT 'Pending',
  Game_CreatedAt    DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (GN_ID) REFERENCES GameNight(GN_ID),
  CONSTRAINT uq_game_number UNIQUE (GN_ID, Game_Number),
  CONSTRAINT chk_game_number CHECK (Game_Number BETWEEN 1 AND 5)
);

CREATE TABLE Cards (
  -- Core fields
  Card_ID         INT          NOT NULL AUTO_INCREMENT PRIMARY KEY,
  Game_ID         INT          NOT NULL,
  Card_Date_Create DATETIME    NOT NULL DEFAULT CURRENT_TIMESTAMP,
  Card_PlayerName VARCHAR(100) NULL,
  Card_PlayerEmail VARCHAR(150) NULL,
  PlayCount       INT          NOT NULL DEFAULT 0,
  Card_IsWinner   TINYINT(1)   NOT NULL DEFAULT 0,
  Card_SeedKey    VARCHAR(64)  NULL,
  Card_PrintedAt  DATETIME     NULL,
  -- 25 Bingo Squares (B=col1, I=col2, N=col3, G=col4, O=col5 / rows 1-5)
  Sq_B1 INT NULL, Sq_B2 INT NULL, Sq_B3 INT NULL, Sq_B4 INT NULL, Sq_B5 INT NULL,
  Sq_I1 INT NULL, Sq_I2 INT NULL, Sq_I3 INT NULL, Sq_I4 INT NULL, Sq_I5 INT NULL,
  Sq_N1 INT NULL, Sq_N2 INT NULL, Sq_N3 INT NULL, Sq_N4 INT NULL, Sq_N5 INT NULL,  -- Sq_N3 = FREE SPACE
  Sq_G1 INT NULL, Sq_G2 INT NULL, Sq_G3 INT NULL, Sq_G4 INT NULL, Sq_G5 INT NULL,
  Sq_O1 INT NULL, Sq_O2 INT NULL, Sq_O3 INT NULL, Sq_O4 INT NULL, Sq_O5 INT NULL,
  FOREIGN KEY (Game_ID) REFERENCES Game(Game_ID),
  FOREIGN KEY (Sq_B1) REFERENCES Songs(Song_ID),
  /* ... repeat FK for all 24 active squares ... */
  FOREIGN KEY (Sq_O5) REFERENCES Songs(Song_ID)
);
